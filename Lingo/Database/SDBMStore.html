def _set(key, val)
  if val.bytesize > MAX_LENGTH
    warn "Warning: Entry `#{key}' (#{@srcfile}) too long for SDBM. Truncating..."
    val = val.byteslice(0, MAX_LENGTH)
  end

  super
end
