def initialize(id, lingo)
  super

  gen = Regexp.escape(GENDER_SEPARATOR)
  sep = Regexp.escape(@sep ||= DEFAULT_SEPARATOR)

  w, a = '\w%1$s(?:\|\w%1$s)*', '[+]?'
  wc   = "##{w % a}(?:#{gen}#{w % ''})?"

  @pat = /^(#{@wrd})#{sep}((?:#{@wrd}#{wc})+)$/
end

def convert_line(line, key, val)
  values = []

  val.strip.scan(/(\S.*?)\s*#(\S+)/) { |k, v|
    v, f = v.split('.')

    v.split('|').product(f ? f.split('|') : [nil]) { |w, g|
      values << "#{k}##{w}##{g}"
    }
  }

  [key.strip, values]
end
